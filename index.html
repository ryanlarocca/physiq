⚡ One-Time Setup
Physiq needs a Supabase database. Run this SQL once to create it, then reload.
SQL to run 
CREATE TABLE IF NOT EXISTS weight_entries (
  id bigserial PRIMARY KEY,
  date text NOT NULL,
  weight float NOT NULL,
  user_id uuid REFERENCES auth.users(id),
  created_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS macro_entries (
  id bigserial PRIMARY KEY,
  date text NOT NULL,
  calories float NOT NULL DEFAULT 0,
  protein float NOT NULL DEFAULT 0,
  carbs float NOT NULL DEFAULT 0,
  fat float NOT NULL DEFAULT 0,
  food_name text NOT NULL DEFAULT '',
  time_logged text,
  user_id uuid REFERENCES auth.users(id),
  created_at timestamptz DEFAULT now()
);

ALTER TABLE weight_entries ENABLE ROW LEVEL SECURITY;
ALTER TABLE macro_entries ENABLE ROW LEVEL SECURITY;

-- Add user_id to existing tables (if already created without it)
ALTER TABLE weight_entries ADD COLUMN IF NOT EXISTS user_id uuid REFERENCES auth.users(id);
ALTER TABLE macro_entries ADD COLUMN IF NOT EXISTS user_id uuid REFERENCES auth.users(id);

-- Drop old open policies, add user-scoped policies
DROP POLICY IF EXISTS "allow_all_weight" ON weight_entries;
DROP POLICY IF EXISTS "allow_all_macros" ON macro_entries;

CREATE POLICY "user_weight" ON weight_entries FOR ALL TO authenticated
  USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id);

CREATE POLICY "user_macros" ON macro_entries FOR ALL TO authenticated
  USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id);

CREATE INDEX IF NOT EXISTS idx_weight_date ON weight_entries(date);
CREATE INDEX IF NOT EXISTS idx_macro_date ON macro_entries(date);
CREATE INDEX IF NOT EXISTS idx_weight_user ON weight_entries(user_id);
CREATE INDEX IF NOT EXISTS idx_macro_user ON macro_entries(user_id);

-- Quick Add Presets (enables cloud sync of quick-add buttons)
CREATE TABLE IF NOT EXISTS quick_add_presets (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid NOT NULL REFERENCES auth.users(id),
  label text NOT NULL,
  calories numeric DEFAULT 0,
  protein numeric DEFAULT 0,
  carbs numeric DEFAULT 0,
  fat numeric DEFAULT 0,
  sort_order integer DEFAULT 0,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE quick_add_presets ENABLE ROW LEVEL SECURITY;
CREATE POLICY "user_presets" ON quick_add_presets FOR ALL TO authenticated
  USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id);
GRANT ALL ON quick_add_presets TO authenticated;
